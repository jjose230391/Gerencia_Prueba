router.beforeEach((to, from, next) => {
  if (to.meta.requiresAuth) {
    const authUser = JSON.parse(window.localStorage.getItem('authUser'))
    if (authUser) {
      next({name: from.name})
    } else {
      next({path: '/login'})
    }
  }
  next()
