  if (JSON.parse(window.localStorage.getItem('authUser') !== null)) {
    const tokenData = JSON.parse(window.localStorage.getItem('authUser'))
    const headers = {
      'Authorization': 'Bearer ' + tokenData.access_token
    }
    return headers
  }
