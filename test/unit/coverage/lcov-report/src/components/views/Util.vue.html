  methods: {
    fetchData () {
      let that = this
      axios.get(`${Urlmain}api/v1/util/`, {headers: getHeader()})
                .then(response => {
                  that.util = response.data.results
                  if (that.util.length === 0) {
                    let body = new FormData()
                    body.append('nombre_campo', 'name')
                    body.append('valor', true)
                    axios.post(`${Urlmain}api/v1/util/`, body, {headers: getHeader()})
                  }
                  if (that.util.length > 10) {
                    this.$nextTick(() => {
                      $('#example1').DataTable()
                    })
                  }
                })
    },
    del (id) {
      this.$Modal.confirm({
        title: 'Delete',
        closable: true,
        content: '<p>Do you want to delete this item?</p>',
        onOk: () => {
          axios.delete(`${Urlmain}api/v1/util/${id}/`, {headers: getHeader()})
          .then(response => {
            this.$Notice.success({
              title: 'Confirmed',
              desc: 'The item was successfully deleted.'
            })
            this.fetchData()
          })
        }
      })
