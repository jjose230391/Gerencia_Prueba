<script>
require('moment')
import datepicker from 'vue-date-picker'
import axios from 'axios'
import {Urlmain, getHeader} from '../../config/index'
export default {
  name: 'CoursesAdd',
  components: { datepicker },
  data () {
    return {
      alert_success: true,
      alergia_error: false,
      course_error: false,
      course_hora_fin_error: false,
      course_fecha_error: false,
      alert_error: true,
      show: true,
      data: '',
      selectFile: null,
      disabledDate (date) {
        return date && date.valueOf() < Date.now() - 86400000
      },
      coursesadd: {
        titulo: '',
        descripcion: '',
        lugar: '',
        hora_inicio: '',
        hora_fin: '',
        fecha: '',
        responsable: '',
        capacidad: '',
        alergia: '',
        imagen: '',
        creditos: '',
        tipo: '',
        duracion: ''
      }
    }
  },
  computed: {
    datetime () {
      return new Date()
    }
  },
  methods: {
    handleChange (date) {
      this.coursesadd.fecha = date
    },
    clearInput (vueModel) {
      vueModel = ''
    },
    onFileSelect (event) {
      if (event.target.files[0] !== '') {
        this.selectFile = event.target.files[0]
      }
    },
    add (coursesadd) {
      let that = this
      let body = new FormData()
      body.append('titulo', coursesadd.titulo)
      body.append('descripcion', coursesadd.descripcion)
      body.append('lugar', coursesadd.lugar)
      body.append('hora_inicio', coursesadd.hora_inicio)
      body.append('hora_fin', coursesadd.hora_fin)
      body.append('capacidad', coursesadd.capacidad)
      body.append('alergia', coursesadd.alergia)
      body.append('responsable', coursesadd.responsable)
      body.append('creditos', coursesadd.creditos)
      body.append('tipo', coursesadd.tipo)
      body.append('fecha', coursesadd.fecha)
      var hora2 = (coursesadd.hora_inicio).split(':')
      var hora1 = (coursesadd.hora_fin).split(':')
      var t1 = new Date()
      var t2 = new Date()
      t1.setHours(hora1[0], hora1[1], hora1[2])
      t2.setHours(hora2[0], hora2[1], hora2[2])
      t1.setHours(t1.getHours() - t2.getHours(), t1.getMinutes() - t2.getMinutes(), t1.getSeconds() - t2.getSeconds())
      var tiempo = (t1.getHours() ? t1.getHours() + (t1.getHours() > 1 ? ':' : ':') : '00') + (t1.getMinutes() ? '' + t1.getMinutes() + (t1.getMinutes() > 1 ? ':' : ':') : '') + (t1.getSeconds() ? (t1.getHours() || t1.getMinutes() ? '' : ':') + t1.getSeconds() + (t1.getSeconds() > 1 ? ' ' : ' ') : '00')
      body.append('duracion', tiempo)
      if (that.selectFile !== null) {
        body.append('imagen', that.selectFile)
      }
      axios.post(`${Urlmain}api/v1/curso_taller/`, body, {headers: getHeader()})
                .then(response => {
                  if (response.status === 201) {
                    this.$Notice.success({
                      title: 'Confirmed',
                      desc: 'The item was successfully add.'
                    })
                    this.$router.push({name: 'Courses'})
                  }
                })
                .catch((error) => {
                  if (error.response.request.status === 400) {
                    this.$Notice.error({
                      title: 'Error',
                      desc: 'Verify item.'
                    })
                    if (error.response.data.hora_fin) {
                      this.course_hora_fin_error = true
                    }
                    if (error.response.data.fecha) {
                      this.course_fecha_error = true
                    }
                  }
                })
    }
  },
  watch: {
    'coursesadd.hora_fin': function () {
      let that = this
      if (that.coursesadd.hora_fin !== '') {
        that.course_hora_fin_error = false
      }
    },
    'coursesadd.fecha': function () {
      let that = this
      if (that.coursesadd.fecha !== '') {
        that.course_fecha_error = false
      }
    }
  }
}
</script>
