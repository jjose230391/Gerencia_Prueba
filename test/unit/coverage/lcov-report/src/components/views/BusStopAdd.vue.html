<script>
require('moment')
import datepicker from 'vue-date-picker'
import axios from 'axios'
import {Urlmain, getHeader} from '../../config/index'
export default {
  name: 'BusStopAdd',
  components: { datepicker },
  created: function () {
    this.fetchData()
  },
  data () {
    return {
      busstop_time_error: false,
      stopadd: {
        nombre: ''
      },
      busstopadd: {
        parada: '',
        omnibus: '',
        hora: '',
        ida: ''
      },
      busadd: {
        capacidad: '',
        ruta: '',
        cantidad_reservaciones: ''
      },
      bus: '',
      modal2: false,
      modal1: false,
      busstop: '',
      ruta: '',
      parade: ''
    }
  },
  computed: {
    datetime () {
      return new Date()
    }
  },
  methods: {
    clearInput (vueModel) {
      vueModel = ''
    },
    cancel () {
      this.modal2 = false
      this.modal1 = false
    },
    fetchData () {
      let that = this
      axios.get(`${Urlmain}api/v1/omnibus/`)
                .then(response => {
                  that.bus = response.data.results
                })
      axios.get(`${Urlmain}api/v1/parada/`)
                .then(response => {
                  that.busstop = response.data.results
                })
    },
    add (busstopadd) {
      let that = this
      let body = new FormData()
      axios.get(`${Urlmain}api/v1/omnibus/?omnibus_ruta=${busstopadd.omnibus}`)
                .then(response => {
                  that.ruta = response.data.results
                  axios.get(`${Urlmain}api/v1/parada/?parada_nombre=${busstopadd.parada}`)
                .then(response => {
                  that.parade = response.data.results
                  body.append('parada', that.parade[0].id)
                  body.append('omnibus', that.ruta[0].id)
                  body.append('hora', busstopadd.hora)
                  if (busstopadd.ida === 'Go') {
                    body.append('ida', true)
                  } else {
                    body.append('ida', false)
                  }
                  axios.post(`${Urlmain}api/v1/parada_omnibus/`, body, {headers: getHeader()})
                .then(response => {
                  if (response.status === 201) {
                    this.$Notice.success({
                      title: 'Confirmed',
                      desc: 'The item was successfully add.'
                    })
                    this.$router.push({name: 'BusStop'})
                  }
                })
                .catch((error) => {
                  if (error.response.request.status === 400) {
                    this.$Notice.error({
                      title: 'Error',
                      desc: 'Verify item.'
                    })
                    if (error.response.data.hora) {
                      this.busstop_time_error = true
                    }
                  }
                })
                })
                })
    },
    addOmnibus (bussadd) {
      let body = new FormData()
      body.append('capacidad', bussadd.capacidad)
      body.append('ruta', bussadd.ruta)
      body.append('cantidad_reservaciones', bussadd.cantidad_reservaciones)
      axios.post(`${Urlmain}api/v1/omnibus/`, body, {headers: getHeader()})
                .then(response => {
                  if (response.status === 201) {
                    this.$Notice.success({
                      title: 'Confirmed',
                      desc: 'The item was successfully add.'
                    })
                    this.modal2 = false
                    this.fetchData()
                  }
                })
    },
    addStopBus (stopadd) {
      let body = new FormData()
      body.append('nombre', stopadd.nombre)
      axios.post(`${Urlmain}api/v1/parada/`, body, {headers: getHeader()})
                .then(response => {
                  if (response.status === 201) {
                    this.$Notice.success({
                      title: 'Confirmed',
                      desc: 'The item was successfully add.'
                    })
                    this.modal1 = false
                    this.fetchData()
                  }
                })
    }
  },
  watch: {
    'busstopadd.hora': function () {
      let that = this
      if (that.busstopadd.hora !== '') {
        that.busstop_time_error = false
      }
    }
  }
}
</script>
