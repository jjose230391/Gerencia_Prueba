<script>
import $ from 'jquery'
import axios from 'axios'
import {Urlmain, getHeader} from '../../config/index'
require('datatables.net')
require('datatables.net-bs')
 
export default {
  name: 'Courses',
  created: function () {
    this.fetchData()
  },
  mounted() {
    this.$nextTick(() => {
      $('#example1').DataTable()
    })
  },
  data () {
    return {
      alert_success: true,
      courses: '',
      select: []
    }
  },
  methods: {
    fetchData () {
      let that = this
      axios.get(`${Urlmain}api/v1/curso_taller/`)
                .then(response => {
                  that.courses = response.data.results
                  this.$nextTick(() => {
                    $('#example1').DataTable()
                  })
                })
    },
    del (id) {
      this.$Modal.confirm({
        title: 'Delete',
        closable: true,
        content: '<p>Do you want to delete this item?</p>',
        onOk: () => {
          axios.delete(`${Urlmain}api/v1/curso_taller/${id}/`, {headers: getHeader()})
          .then(response => {
            this.$Notice.success({
              title: 'Confirmed',
              desc: 'The item was successfully deleted.'
            })
            this.fetchData()
          })
        }
      })
    },
    delet () {
      this.$Modal.confirm({
        title: 'DeleteSelect',
        closable: true,
        content: '<p>Do you want to delete select item?</p>',
        onOk: () => {
          for (let i = 0; i < this.select.length; i++) {
            axios.delete(`${Urlmain}api/v1/actividad/${this.select[i]}/`, {headers: getHeader()})
          .then(response => {
            this.fetchData()
          })
          }
        }
      })
    }
  },
  watch: {
    'select': function () {
    }
  }
}
</script>
