<script>
require('moment')
import datepicker from 'vue-date-picker'
import axios from 'axios'
import {Urlmain, getHeader} from '../../config/index'
export default {
  name: 'UserEdit',
  props: ['id'],
  components: { datepicker },
  created: function () {
    this.DataEdit()
  },
  data () {
    return {
      phone_number_error: false,
      email_error: false,
      code_error: false,
      useredit: {
        first_name: '',
        last_name: '',
        phone_number: '',
        alergias: '',
        acompannante: '',
        tax_number: '',
        email: '',
        code: '',
        password: '',
        is_superuser: '',
        client_id: ''
      }
    }
  },
  computed: {
    datetime () {
      return new Date()
    }
  },
  methods: {
    clearInput (vueModel) {
      vueModel = ''
    },
    DataEdit () {
      let that = this
      axios.get(`${Urlmain}api/v1/usuario/${that.id}/`, { headers: getHeader() })
                .then(response => {
                  that.useredit = response.data
                })
    },
    edit (useredit) {
      let that = this
      let body = new FormData()
      body.append('first_name', useredit.first_name)
      body.append('last_name', useredit.last_name)
      body.append('alergias', useredit.alergias)
      body.append('email', useredit.email)
      body.append('password', useredit.password)
      body.append('is_superuser', useredit.is_superuser)
      body.append('client_id', useredit.client_id)
      body.append('code', useredit.code)
      body.append('tax_number', useredit.tax_number)
      body.append('acompannante', useredit.acompannante)
      body.append('phone_number', useredit.phone_number)
      axios.patch(`${Urlmain}api/v1/usuario/${that.id}/`, body, {headers: getHeader()})
                .then(response => {
                  if (response.status === 200) {
                    this.$Notice.success({
                      title: 'Confirmed',
                      desc: 'The item was successfully modified.'
                    })
                    this.$router.push({name: 'User'})
                  }
                })
                .catch((error) => {
                  if (error.response.request.status === 400) {
                    this.$Notice.error({
                      title: 'Error',
                      desc: 'Verify item.'
                    })
                    if (error.response.data[0] === 'This code has already been assigned') {
                      this.code_error = true
                    }
                    if (error.response.data.phone_number) {
                      this.phone_number_error = true
                    }
                    if (error.response.data.email) {
                      this.email_error = true
                    }
                  }
                })
    }
  },
  watch: {
    'useredit.code': function () {
      let that = this
      if (that.useredit.code !== '') {
        that.code_error = false
      }
    },
    'useredit.email': function () {
      let that = this
      if (that.useredit.email !== '') {
        that.email_error = false
      }
    },
    'useredit.phone_number': function () {
      let that = this
      if (that.useredit.phone_number !== '') {
        that.phone_number_error = false
      }
    }
  }
}
</script>
