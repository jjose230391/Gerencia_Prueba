<script>
require('moment')
import datepicker from 'vue-date-picker'
import axios from 'axios'
import {Urlmain, getHeader} from '../../config/index'
export default {
  name: 'ActivitiesAdd',
  components: { datepicker },
  data () {
    return {
      activities_fecha_error: false,
      activities_hora_fin_error: false,
      activitiesadd: {
        titulo: '',
        responsable: '',
        descripcion: '',
        lugar: '',
        hora_inicio: '',
        hora_fin: '',
        fecha: '',
        tipo: '',
        capacidad: '',
        duracion: ''
      }
    }
  },
  computed: {
    datetime () {
      return new Date()
    }
  },
  methods: {
    clearInput (vueModel) {
      vueModel = ''
    },
    handleChange (date) {
      this.activitiesadd.fecha = date
    },
    add (activitiesadd) {
      let body = new FormData()
      body.append('titulo', activitiesadd.title)
      body.append('descripcion', activitiesadd.descripcion)
      body.append('lugar', activitiesadd.lugar)
      body.append('hora_inicio', activitiesadd.hora_inicio)
      body.append('hora_fin', activitiesadd.hora_fin)
      body.append('capacidad', activitiesadd.capacidad)
      body.append('tipo', activitiesadd.tipo)
      body.append('responsable', activitiesadd.responsable)
      body.append('fecha', activitiesadd.fecha)
      var hora2 = (activitiesadd.hora_inicio).split(':')
      var hora1 = (activitiesadd.hora_fin).split(':')
      var t1 = new Date()
      var t2 = new Date()
      t1.setHours(hora1[0], hora1[1], hora1[2])
      t2.setHours(hora2[0], hora2[1], hora2[2])
      t1.setHours(t1.getHours() - t2.getHours(), t1.getMinutes() - t2.getMinutes(), t1.getSeconds() - t2.getSeconds())
      var tiempo = (t1.getHours() ? t1.getHours() + (t1.getHours() > 1 ? ':' : ':') : '00') + (t1.getMinutes() ? '' + t1.getMinutes() + (t1.getMinutes() > 1 ? ':' : ':') : '') + (t1.getSeconds() ? (t1.getHours() || t1.getMinutes() ? '' : ':') + t1.getSeconds() + (t1.getSeconds() > 1 ? ' ' : ' ') : '00')
      body.append('duracion', tiempo)
      axios.post(`${Urlmain}api/v1/actividad/`, body, {headers: getHeader()})
                .then(response => {
                  if (response.status === 201) {
                    this.$Notice.success({
                      title: 'Confirmed',
                      desc: 'The item was successfully add.'
                    })
                    this.$router.push({name: 'Activities'})
                  }
                })
                .catch((error) => {
                  if (error.response.request.status === 400) {
                    this.$Notice.error({
                      title: 'Error',
                      desc: 'Verify item.'
                    })
                    if (error.response.data.hora_fin) {
                      this.activities_hora_fin_error = true
                    }
                    if (error.response.data.fecha) {
                      this.activities_fecha_error = true
                    }
                  }
                })
    }
  },
  watch: {
    'activitiesadd.hora_fin': function () {
      let that = this
      if (that.activitiesadd.hora_fin !== '') {
        that.activities_hora_fin_error = false
      }
    },
    'activitiesadd.fecha': function () {
      let that = this
      if (that.activitiesadd.fecha !== '') {
        that.activities_fecha_error = false
      }
    }
  }
}
</script>
